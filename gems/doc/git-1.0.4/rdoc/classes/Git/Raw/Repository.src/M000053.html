      def log(sha, count = 30)
        output = ''
        i = 0

        while sha && (i < count) do
          o = get_raw_object_by_sha1(sha)
          c = Git::Raw::Object.from_raw(o)
          
          output += "commit #{sha}\n"
          output += o.content + "\n"

          sha = c.parent.first
          i += 1
        end
        
        output
      end
